# Listę zadań z tabeli ISSUESS z dołączonymi do nich nazwami list zadań z tabeli ISSUESLISTS
SELECT * FROM ISSUESLIST;
SELECT * FROM ISSUES;
SELECT * FROM USERS;

SELECT ISSUESLIST.NAME, ISSUES.SUMMARY, ISSUES.DESCRIPTION
FROM ISSUES
         INNER JOIN ISSUESLIST ON ISSUES.ISSUESLISTS_ID = ISSUESLIST.ID
ORDER BY ISSUESLIST.NAME;


SELECT IL.NAME, I.SUMMARY, I.DESCRIPTION
FROM ISSUES I
         JOIN ISSUESLIST IL ON I.ISSUESLISTS_ID = IL.ID
ORDER BY IL.NAME;

# Listę zadań z tabeli ISSUESS z przypisanymi do nich użytkownikami z tabeli USERS
SELECT * FROM ISSUESLIST;
SELECT * FROM ISSUES;
SELECT * FROM USERS;

SELECT USERS.LASTNAME, USERS.FIRSTNAME, ISSUES.SUMMARY, ISSUES.DESCRIPTION
FROM ISSUES
INNER JOIN USERS ON ISSUES.USER_ID_ASSIGNEDTO = USERS.ID
ORDER BY SUMMARY;

SELECT U.LASTNAME, U.FIRSTNAME, I.SUMMARY, I.DESCRIPTION FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
ORDER BY I.SUMMARY;

# Imiona i nazwiska użytkowników posiadających zadania, wraz z informacją o ilości zadań, które są do nich przypisane
SELECT * FROM ISSUESLIST;
SELECT * FROM ISSUES;
SELECT * FROM USERS;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASK_NUMBER
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO
ORDER BY U.LASTNAME;
